### Task 12: Add TTL to exercise selection singleton
**Priority: Medium | Files: `src/utils/exerciseSelection.ts`**

Add a timestamp when setting the pending exercise, and expire after 30 seconds:
```typescript
let pendingExercise: { id: number; name: string } | null = null;
let pendingTimestamp: number = 0;
const TTL_MS = 30_000;

export function setPendingExercise(exercise: { id: number; name: string }) {
  pendingExercise = exercise;
  pendingTimestamp = Date.now();
}

export function consumePendingExercise() {
  if (pendingExercise && (Date.now() - pendingTimestamp) < TTL_MS) {
    const ex = pendingExercise;
    pendingExercise = null;
    return ex;
  }
  pendingExercise = null;
  return null;
}
```

**Checkpoint:** Exercise selection still works normally. Stale selections expire.
