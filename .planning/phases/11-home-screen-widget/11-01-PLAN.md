# Phase 11: Home Screen Widget & Quick Actions

## Objective
Add Android home screen widget showing workout streak and last workout summary, implement app icon quick actions (both platforms) for starting workouts from saved templates, and trigger widget/action updates on workout events.

> **Scope note:** iOS home screen widget is deferred. WidgetKit requires native Swift/SwiftUI code (via `@bacons/apple-targets`) which is a significant departure from the project's JavaScript-only approach. The official `expo-widgets` package (SDK 55+) will enable JS-based iOS widgets when the project upgrades from SDK 54. Android widget and cross-platform quick actions are included in this plan.

### Task 2: Quick actions — dynamic template-based app icon shortcuts
**What:** Implement app icon shortcuts that appear when users long-press the app icon. Shortcuts are dynamically generated from saved workout templates (up to 3) plus a generic "Start Workout" action. Tapping a template shortcut opens ActiveWorkout with that template pre-loaded. Works on both iOS and Android.
**How:**
- **Create `src/utils/quickActions.ts`:**
  ```typescript
  import * as QuickActions from 'expo-quick-actions';
  import { Platform } from 'react-native';
  import { getAllTemplates } from '../database/services';

  export function refreshQuickActions() {
    const templates = getAllTemplates();
    const items: QuickActions.Action[] = [];

    // Add up to 3 most recent templates as shortcuts
    const topTemplates = templates.slice(0, 3);
    for (const template of topTemplates) {
      items.push({
        id: `template-${template.id}`,
        title: template.name,
        subtitle: template.split_label,
        icon: Platform.OS === 'ios' ? 'symbol:dumbbell.fill' : undefined,
        params: { templateId: String(template.id) },
      });
    }

    // Generic start workout action (always last)
    items.push({
      id: 'start-workout',
      title: 'Start Workout',
      subtitle: 'Choose a split',
      icon: Platform.OS === 'ios' ? 'symbol:plus.circle.fill' : undefined,
    });

    QuickActions.setItems(items);
  }
  ```
  `refreshQuickActions()` is safe to call anytime — replaces all existing items.
- **Create `src/utils/quickActionHandler.ts`:**
  ```typescript
  import type { Action } from 'expo-quick-actions';
  import { navigationRef } from '../../App';
  import { getTemplateWithExercises } from '../database/services';

  export function handleQuickAction(action: Action) {
    if (!navigationRef.isReady()) return;

    if (action.id === 'start-workout') {
      (navigationRef as any).navigate('Log Workout', { screen: 'StartWorkout' });
      return;
    }

    // Template-based launch
    const templateId = action.params?.templateId;
    if (!templateId) return;

    const template = getTemplateWithExercises(Number(templateId));
    if (!template) {
      // Template was deleted — fall back to StartWorkout
      (navigationRef as any).navigate('Log Workout', { screen: 'StartWorkout' });
      return;
    }

    (navigationRef as any).navigate('Log Workout', {
      screen: 'ActiveWorkout',
      params: {
        workoutId: -1,
        muscleGroupId: template.muscle_group_id,
        splitLabel: template.split_label,
        muscleGroupIds: template.muscle_group_ids,
        fromTemplate: {
          templateId: template.id,
          exercises: template.exercises.map((e) => ({
            exerciseId: e.exercise_id,
            exerciseName: e.exercise_name,
            defaultSets: e.default_sets,
          })),
        },
      },
    });
  }
  ```
  Uses `any` cast on navigationRef for nested navigator navigation (matches existing codebase pattern with `useNavigation<any>()`).
- **Integrate in `App.tsx`:**
  - Add imports:
    ```typescript
    import * as QuickActions from 'expo-quick-actions';
    import { refreshQuickActions } from './src/utils/quickActions';
    import { handleQuickAction } from './src/utils/quickActionHandler';
    ```
  - In `AppContent` function, add quick action callback hook (before the return):
    ```typescript
    // Handle quick actions when app is already running (warm start)
    QuickActions.useQuickActionCallback((action) => {
      handleQuickAction(action);
    });

    // Handle quick action that launched the app (cold start)
    useEffect(() => {
      if (QuickActions.initial) {
        const timer = setTimeout(() => handleQuickAction(QuickActions.initial!), 300);
        return () => clearTimeout(timer);
      }
    }, []);
    ```
    Both hooks go in `AppContent` (not `App`) because `AppContent` is only rendered when `isReady` is true (DB initialized) and wraps `NavigationContainer`. The 300ms delay on cold start ensures the navigation container is fully mounted before navigating.
  - Add `useEffect` import if not already present.
  - In `attemptInit()` (in `App` component), after `syncNotificationSchedules()`, add:
    ```typescript

    ```
- **Refresh on template changes — `src/screens/StartWorkoutScreen.tsx`:**
  - Add import: `import { refreshQuickActions } from '../utils/quickActions';`
  - In the existing `useFocusEffect`, after the three `set*` calls (line 49-51), add:
    ```typescript

    ```

- **Create `src/widget/handler.ts`:**
  ```typescript
  import React from 'react';
  import { Linking } from 'react-native';
  import type { WidgetTaskHandlerProps } from 'react-native-android-widget';
  import { WorkoutWidget } from './WorkoutWidget';
  import {
    getWeeklyStreak,
    getWorkouts,
    getMonthlyStats,
    getSetting,
  } from '../database/services';

  function formatRelativeDate(dateStr: string): string {
    const today = new Date().toISOString().split('T')[0];
    const yesterday = new Date();
    yesterday.setDate(yesterday.getDate() - 1);
    const yesterdayStr = yesterday.toISOString().split('T')[0];

    if (dateStr === today) return 'Today';
    if (dateStr === yesterdayStr) return 'Yesterday';
    const date = new Date(dateStr + 'T00:00:00');
    return date.toLocaleDateString('en-US', { month: 'short', day: 'numeric' });
  }

  function formatVolume(volume: number): string {
    if (volume >= 10000) {
      return `${(volume / 1000).toFixed(1).replace(/\.0$/, '')}k`;
    }
    return volume.toLocaleString();
  }

  function getWidgetData() {
    try {
      const goalStr = getSetting('weekly_goal', '3');
      const weeklyGoal = parseInt(goalStr, 10) || 3;
      const streak = getWeeklyStreak(weeklyGoal);

      const now = new Date();
      const stats = getMonthlyStats(now.getFullYear(), now.getMonth() + 1);

      const workouts = getWorkouts();
      const lastWorkout = workouts.length > 0 ? workouts[0] : null;

      let lastWorkoutLabel = '';
      let lastWorkoutMeta = '';
      if (lastWorkout) {
        lastWorkoutLabel = lastWorkout.muscle_group_name;
        lastWorkoutMeta = `${formatRelativeDate(lastWorkout.date)} · ${lastWorkout.exercise_count} exercises · ${formatVolume(lastWorkout.total_volume)} lbs`;
      }

      return {
        streak,
        monthlyWorkouts: stats.workoutCount,
        lastWorkoutLabel,
        lastWorkoutMeta,
        hasWorkouts: workouts.length > 0,
      };
    } catch (e) {
      console.warn('Widget data fetch failed:', e);
      return {
        streak: 0,
        monthlyWorkouts: 0,
        lastWorkoutLabel: '',
        lastWorkoutMeta: '',
        hasWorkouts: false,
      };
    }
  }

  export async function widgetTaskHandler(props: WidgetTaskHandlerProps) {
    switch (props.widgetAction) {
      case 'WIDGET_ADDED':
      case 'WIDGET_UPDATE':
      case 'WIDGET_RESIZED':
        const data = getWidgetData();
        props.renderWidget(<WorkoutWidget {...data} />);
        break;

      case 'WIDGET_CLICK':
        if (props.clickAction === 'START_WORKOUT') {
          Linking.openURL('workouttracker://workout');
        } else if (props.clickAction === 'OPEN_APP') {
          Linking.openURL('workouttracker://');
        }
        break;

      case 'WIDGET_DELETED':
        break;
    }
  }
  ```
  The handler runs in a headless JS context. `getDatabase()` (called internally by service functions) uses `openDatabaseSync()` which opens the existing DB file and runs idempotent schema init. This is safe in the headless context — same app process, same filesystem.
- **Register handler in `index.ts`:**
  ```typescript
  import { registerRootComponent } from 'expo';
  import { registerWidgetTaskHandler } from 'react-native-android-widget';
  import { widgetTaskHandler } from './src/widget/handler';
  import App from './App';

  registerWidgetTaskHandler(widgetTaskHandler);
  registerRootComponent(App);
  ```
  `registerWidgetTaskHandler` is a no-op on iOS — safe to call unconditionally. Must be called before `registerRootComponent` so it's available when Android triggers a headless widget task.
**Output:** `src/widget/WorkoutWidget.tsx` (widget UI component), `src/widget/handler.ts` (task handler with SQLite data fetching). Handler registered in `index.ts`. Widget displays streak, monthly workouts, last workout summary, and "Start Workout" button with deep link.
