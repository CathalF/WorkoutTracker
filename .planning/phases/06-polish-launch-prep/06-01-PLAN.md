- Create `src/theme/ThemeContext.tsx`:
  ```typescript
  import { createContext, useContext, ReactNode } from 'react';
  import { useColorScheme } from 'react-native';
  import { lightColors, darkColors, ThemeColors } from './colors';

  const ThemeContext = createContext<ThemeColors>(lightColors);

  export function ThemeProvider({ children }: { children: ReactNode }) {
    const scheme = useColorScheme();
    const colors = scheme === 'dark' ? darkColors : lightColors;
    return (
      <ThemeContext.Provider value={colors}>
        {children}
      </ThemeContext.Provider>
    );
  }

  export function useTheme(): ThemeColors {
    return useContext(ThemeContext);
  }
  ```
- Create `src/theme/index.ts` that re-exports:
  ```typescript
  export { lightColors, darkColors } from './colors';
  export type { ThemeColors } from './colors';
  export { ThemeProvider, useTheme } from './ThemeContext';
  ```
- Update `App.tsx`:

### Task 2: Migrate all screens and navigation to use theme
**What:** Replace every hardcoded color value across all screens and navigation files with theme values from `useTheme()`. Screens should dynamically style based on the active theme.
**How:**
- **Strategy:** In each screen file:
  1. Add `import { useTheme } from '../theme';` at the top
  2. Add `const colors = useTheme();` at the start of the component function
  3. Move color-dependent styles from `StyleSheet.create()` into inline `useMemo`-based styles or a `useStyles(colors)` pattern. The recommended pattern is to convert `StyleSheet.create()` to a function `createStyles(colors: ThemeColors)` called with `useMemo`:
     ```typescript
     const styles = useMemo(() => createStyles(colors), [colors]);
     // ...
     const createStyles = (colors: ThemeColors) => StyleSheet.create({ ... });
     ```
  4. Replace all hardcoded color strings with `colors.xxx` references
- **Files to update (in order):**
  1. `src/navigation/AppNavigator.tsx` — update `tabBarActiveTintColor` and `tabBarInactiveTintColor` to use theme colors

### Task 3: Add error boundary and improve loading/error states
**What:** Add a React error boundary component that catches render errors, improve the database initialization failure UX, and add a proper loading spinner to WorkoutDetailScreen.
**How:**
- Create `src/components/ErrorBoundary.tsx`:
  - Class component (error boundaries require class components)
  - State: `{ hasError: boolean, error: Error | null }`
  - `static getDerivedStateFromError(error)` → sets hasError true
  - `componentDidCatch(error, info)` → console.error for logging
  - Fallback UI: centered screen with:
    - `alert-circle-outline` icon (64px, red)
    - "Something went wrong" title
    - Error message in secondary text
    - "Try Again" button that calls `this.setState({ hasError: false })` to retry
  - Use static colors (not theme) since error boundary wraps the theme provider
- Update `App.tsx`:
  - Wrap entire app in `<ErrorBoundary>`
  - Add an error state: `const [initError, setInitError] = useState(false);`
  - In the catch block, set `setInitError(true)` in addition to console.error
  - If `initError` is true, show a dedicated error screen:
    - `cloud-offline-outline` icon, "Database Error" title
    - "Failed to initialize the database. Please restart the app." subtitle
    - "Retry" button that calls `initializeDatabase()` again
- Update `src/screens/WorkoutDetailScreen.tsx`:
  - Add `const [isLoading, setIsLoading] = useState(true);` state
  - In `loadData`, set `setIsLoading(true)` before and `setIsLoading(false)` after
  - When `isLoading` is true, show an `ActivityIndicator` instead of "Workout not found"
  - Only show "Workout not found" when `isLoading === false && workout === null`
**Output:** ErrorBoundary wraps the app, DB init failure shows retry UI, WorkoutDetailScreen has proper loading spinner
